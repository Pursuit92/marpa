#![allow(unused)]

pub struct Error(i32, Option<String>);

impl Error {
    pub fn get_code(&self) -> i32 {
        self.0
    }
}

impl ::std::error::Error for Error {
    fn description(&self) -> &str {
        match self.0 {
            i if i >= 0 => MARPA_ERROR_DESCRIPTION[self.0 as usize].2,
            _ => match self.1 {
                Some(ref s) => s,
                _ => "undefined error"
            }
        }
    }
}

impl ::std::fmt::Display for Error {
    fn fmt(&self, f: &mut ::std::fmt::Formatter) -> ::std::fmt::Result {
        use std::error::Error;
        write!(f, "{}", self.description())
    }
}

impl ::std::fmt::Debug for Error {
    fn fmt(&self, f: &mut ::std::fmt::Formatter) -> ::std::fmt::Result {
        write!(f, "{}", self)
    }
}

impl From<i32> for Error {
    fn from(other: i32) -> Error {
        match other {
            i if i >= 0 => Error(i, None),
            i => Error(i, Some(format!("undefined error: {}", i))),
        }
    }
}

impl<'a> From<&'a str> for Error {
    fn from(other: &str) -> Error {
        other.into()
    }
}

impl From<String> for Error {
    fn from(other: String) -> Error {
        Error(-1, Some(other))
    }
}

impl ::std::ops::Deref for Error {
    type Target = str;

    fn deref(&self) -> &str {
        use ::std::error::Error;
        self.description()
    }
}
